Commandes Rapides - Copier/Coller

Des blocs de code prets a copier pour l’examen. Clique sur l’icone de copie en haut a droite de chaque bloc.

1 Demarrage de Session (TOUJOURS en premier)

library(tidyverse)
# Si besoin de definir le repertoire de travail
setwd("/chemin/vers/mon/dossier")

2 Importer des Donnees

# Importer un fichier CSV
df <- read_csv("mon_fichier.csv")
# Importer un CSV avec separateur point-virgule (format europeen)
df <- read_csv2("mon_fichier.csv")
# Importer un fichier Excel
library(readxl)
df <- read_excel("mon_fichier.xlsx")

3 Explorer les Donnees (premiere chose a faire apres import)

# Vue rapide : combien de lignes/colonnes, quels types ?
dim(df)       # Nombre de lignes et colonnes
names(df)     # Noms des colonnes
str(df)       # Structure complete (types, apercu)
summary(df)   # Stats descriptives de chaque colonne
head(df)      # 6 premieres lignes
# Combien de valeurs manquantes ?
sum(is.na(df))                  # Total de NA dans tout le dataset
colSums(is.na(df))              # NA par colonne
# Combien de valeurs uniques dans une colonne ?
n_distinct(df$ma_colonne)
unique(df$ma_colonne)           # Voir les valeurs uniques

4 Manipulation avec dplyr - Templates

4.1 Selectionner des colonnes

df %>% select(col1, col2, col3)
df %>% select(-colonne_a_exclure)
df %>% select(starts_with("IG"))

4.2 Filtrer des lignes

df %>% filter(age > 25)
df %>% filter(genre == "Femme")
df %>% filter(note >= 4 & branche == "Psycho")
df %>% filter(pays %in% c("Suisse", "France", "Allemagne"))
df %>% filter(!is.na(prix))     # Exclure les valeurs manquantes

4.3 Creer / modifier une colonne

df %>% mutate(nouvelle_col = col1 / col2)
df %>% mutate(reussi = note >= 4)
# Creer une colonne conditionnelle (si/sinon)
df %>% mutate(
  categorie = case_when(
    note >= 5.5  ~ "Tres bien",
    note >= 5.0  ~ "Bien",
    note >= 4.0  ~ "Suffisant",
    TRUE         ~ "Insuffisant"
  )
)

4.4 Trier

df %>% arrange(note)              # Croissant
df %>% arrange(desc(note))        # Decroissant

4.5 Compter les occurrences

df %>% count(ma_variable)
df %>% count(ma_variable, sort = TRUE)   # Tri decroissant

4.6 Resumer par groupe (le plus courant a l’examen)

df %>%
  group_by(variable_categorielle) %>%
  summarize(
    nombre    = n(),
    moyenne   = mean(variable_num, na.rm = TRUE),
    mediane   = median(variable_num, na.rm = TRUE),
    ecart_type = sd(variable_num, na.rm = TRUE),
    minimum   = min(variable_num, na.rm = TRUE),
    maximum   = max(variable_num, na.rm = TRUE)
  ) %>%
  ungroup()

4.7 Chaine complete typique (filtre + groupe + resume + tri)

df %>%
  filter(condition > valeur) %>%
  group_by(categorie) %>%
  summarize(
    n = n(),
    moy = mean(variable, na.rm = TRUE)
  ) %>%
  ungroup() %>%
  arrange(desc(moy))

5 Graphiques ggplot2 - Templates

5.1 Nuage de points (2 variables continues)

ggplot(df, aes(x = var_continue_1, y = var_continue_2)) +
  geom_point() +
  labs(title = "Mon titre", x = "Axe X (unite)", y = "Axe Y (unite)")
# Avec couleur par categorie
ggplot(df, aes(x = var_x, y = var_y, color = var_categorie)) +
  geom_point() +
  labs(title = "Titre", x = "X", y = "Y", color = "Legende")

5.2 Diagramme a barres (frequences d’une categorie)

ggplot(df, aes(x = var_categorielle)) +
  geom_bar() +
  labs(title = "Mon titre", x = "Categorie", y = "Nombre")
# Barres cote a cote par sous-groupe
ggplot(df, aes(x = var_cat_1, fill = var_cat_2)) +
  geom_bar(position = "dodge") +
  labs(title = "Titre", x = "X", y = "Nombre", fill = "Legende")

5.3 Histogramme (distribution d’une variable continue)

ggplot(df, aes(x = var_continue)) +
  geom_histogram(binwidth = 10) +
  labs(title = "Mon titre", x = "Variable (unite)", y = "Frequence")

5.4 Boxplot (comparer distributions entre groupes)

ggplot(df, aes(x = var_categorielle, y = var_continue)) +
  geom_boxplot() +
  labs(title = "Mon titre", x = "Groupe", y = "Variable (unite)")

5.5 Sous-graphiques par categorie (facettes)

# Ajouter cette ligne a n'importe quel graphique
+ facet_wrap(~ var_categorielle)

5.6 Personnalisation rapide (a ajouter avec +)

+ labs(title = "Titre", subtitle = "Sous-titre", x = "Axe X", y = "Axe Y",
       color = "Legende", fill = "Legende", caption = "Source : ...")
+ theme_minimal()           # Theme epure
+ theme_bw()                # Theme noir et blanc
+ theme(legend.position = "none")   # Supprimer la legende
+ scale_fill_viridis_d()    # Palette accessible (daltonisme)
+ scale_color_viridis_d()   # Idem pour les couleurs de points/lignes

6 Statistiques Rapides (hors graphique)

mean(df$col, na.rm = TRUE)      # Moyenne
median(df$col, na.rm = TRUE)    # Mediane
sd(df$col, na.rm = TRUE)        # Ecart-type
min(df$col, na.rm = TRUE)       # Minimum
max(df$col, na.rm = TRUE)       # Maximum
sum(df$col, na.rm = TRUE)       # Somme

7 Conversions de Types

# Convertir en numerique (si une colonne est lue comme texte)
df$col <- as.numeric(df$col)

# Convertir en facteur (variable categorielle)
df$col <- as.factor(df$col)

# Convertir en texte
df$col <- as.character(df$col)

8 Gestion des Valeurs Manquantes (NA)

# Compter les NA
sum(is.na(df$col))

# Supprimer les lignes avec NA dans une colonne
df %>% filter(!is.na(col))

# Supprimer TOUTES les lignes qui ont au moins un NA
df %>% drop_na()

# Remplacer les NA par une valeur
df %>% mutate(col = replace_na(col, 0))
